# C++ 多线程编程

多线程编程是现代C++高性能应用的关键技术。本章将系统介绍线程管理、同步机制、并发模式以及相关的高级特性。

---

## 1. 线程基础

### 核心概念

- **线程**：程序执行的最小单位，多个线程可以并发执行
- **并发**：多个任务在重叠的时间段内执行
- **并行**：多个任务同时执行（需要多核处理器）
- **同步**：协调多个线程对共享资源的访问

### 线程创建与管理

```cpp
#include <iostream>
#include <thread>
#include <vector>
#include <chrono>

using namespace std;

// 方式1：函数指针
void threadFunction(int id) {
    cout << "线程 " << id << " 开始执行" << endl;
    this_thread::sleep_for(chrono::milliseconds(100));
    cout << "线程 " << id << " 结束执行" << endl;
}

// 方式2：Lambda表达式
void lambdaThreadDemo() {
    vector<thread> threads;
    
    for (int i = 0; i < 5; ++i) {
        threads.emplace_back([i]() {
            cout << "Lambda线程 " << i << " 运行中" << endl;
        });
    }
    
    for (auto& t : threads) {
        t.join();
    }
}

// 方式3：类成员函数
class Worker {
public:
    void operator()() const {
        cout << "Worker线程执行" << endl;
    }
    
    void doWork(int value) {
        cout << "处理数据: " << value << endl;
    }
};

int main() {
    // 函数指针
    thread t1(threadFunction, 1);
    t1.join();
    
    // Lambda
    lambdaThreadDemo();
    
    // 函数对象
    thread t2(Worker());
    t2.join();
    
    // 成员函数
    Worker worker;
    thread t3(&Worker::doWork, &worker, 42);
    t3.join();
    
    return 0;
}
```

## 2. 数据竞争与同步

### 数据竞争问题

```cpp
#include <atomic>

void dataRaceDemo() {
    int unsafeCounter = 0;
    atomic<int> safeCounter(0);
    
    // 不安全的计数器
    auto unsafeIncrement = [&]() {
        for (int i = 0; i < 10000; ++i) {
            unsafeCounter++;  // 数据竞争！
        }
    };
    
    // 安全的计数器
    auto safeIncrement = [&]() {
        for (int i = 0; i < 10000; ++i) {
            safeCounter++;  // 原子操作
        }
    };
    
    vector<thread> unsafeThreads, safeThreads;
    
    for (int i = 0; i < 10; ++i) {
        unsafeThreads.emplace_back(unsafeIncrement);
        safeThreads.emplace_back(safeIncrement);
    }
    
    for (auto& t : unsafeThreads) t.join();
    for (auto& t : safeThreads) t.join();
    
    cout << "不安全计数: " << unsafeCounter << " (期望: 100000)" << endl;
    cout << "安全计数: " << safeCounter << " (期望: 100000)" << endl;
}
```

### 互斥锁

```cpp
#include <mutex>

void mutexDemo() {
    mutex mtx;
    int sharedData = 0;
    
    auto lockedIncrement = [&]() {
        for (int i = 0; i < 1000; ++i) {
            lock_guard<mutex> lock(mtx);  // RAII锁
            sharedData++;
        }
    };
    
    vector<thread> threads;
    for (int i = 0; i < 10; ++i) {
        threads.emplace_back(lockedIncrement);
    }
    
    for (auto& t : threads) t.join();
    cout << "共享数据: " << sharedData << " (期望: 10000)" << endl;
}
```

### 锁类型对比

| 锁类型 | 特点 | 适用场景 |
|--------|------|----------|
| `lock_guard` | 简单自动锁，作用域结束时释放 | 简单临界区 |
| `unique_lock` | 可延迟锁定、提前解锁、转移所有权 | 条件变量、灵活锁管理 |
| `scoped_lock` | 可同时锁定多个互斥锁（C++17） | 避免多锁死锁 |
| `shared_lock` | 共享读锁（C++17） | 读写锁场景 |

```cpp
#include <shared_mutex>

void lockTypesDemo() {
    mutex mtx;
    shared_mutex sharedMtx;
    
    // lock_guard：简单自动锁
    {
        lock_guard<mutex> lock(mtx);
        // 临界区代码
    }
    
    // unique_lock：灵活锁管理
    {
        unique_lock<mutex> lock(mtx, defer_lock);  // 延迟锁定
        lock.lock();  // 手动锁定
        // 临界区代码
        lock.unlock();  // 可以提前解锁
    }
    
    // scoped_lock：多锁管理
    {
        mutex mtx1, mtx2;
        scoped_lock lock(mtx1, mtx2);  // 同时锁定两个互斥锁
        // 临界区代码
    }
    
    // shared_lock：读锁
    {
        shared_lock<shared_mutex> readLock(sharedMtx);
        // 只读访问
    }
    
    // 写锁
    {
        unique_lock<shared_mutex> writeLock(sharedMtx);
        // 写入访问
    }
}
```

## 3. 原子操作

### 基本原子类型

```cpp
#include <atomic>

void atomicDemo() {
    // 原子整数
    atomic<int> counter(0);
    counter++;             // 原子自增
    counter.fetch_add(5);  // 原子加5
    int val = counter.load();  // 原子读取
    counter.store(10);     // 原子写入
    
    // 原子布尔
    atomic<bool> flag(false);
    flag.store(true);
    bool expected = false;
    if (flag.compare_exchange_strong(expected, true)) {
        cout << "成功设置标志" << endl;
    }
    
    // 原子指针
    int arr[] = {1, 2, 3};
    atomic<int*> ptr(arr);
    int* old = ptr.exchange(arr + 1);
    cout << "旧指针指向: " << *old << endl;
}
```

### 内存顺序

```cpp
void memoryOrderDemo() {
    atomic<int> x(0), y(0);
    
    // relaxed：无顺序保证
    x.store(1, memory_order_relaxed);
    
    // release-acquire：建立同步关系
    thread t1([&]() {
        x.store(1, memory_order_release);
        y.store(1, memory_order_release);
    });
    
    thread t2([&]() {
        while (y.load(memory_order_acquire) != 1);
        // 保证能看到 x = 1
        cout << "x = " << x.load(memory_order_acquire) << endl;
    });
    
    t1.join();
    t2.join();
    
    // seq_cst（默认）：最强顺序保证
    atomic<int> z(0);
    z.store(1, memory_order_seq_cst);
}
```

### 自旋锁实现

```cpp
class SpinLock {
private:
    atomic_flag flag = ATOMIC_FLAG_INIT;
    
public:
    void lock() {
        while (flag.test_and_set(memory_order_acquire)) {
            // 自旋等待
        }
    }
    
    void unlock() {
        flag.clear(memory_order_release);
    }
};
```

## 4. 条件变量

### 生产者-消费者模式

```cpp
#include <queue>
#include <condition_variable>

template<typename T>
class ThreadSafeQueue {
private:
    queue<T> data;
    mutable mutex mtx;
    condition_variable cond;
    bool stopped = false;
    
public:
    void push(T value) {
        lock_guard<mutex> lock(mtx);
        data.push(move(value));
        cond.notify_one();
    }
    
    optional<T> pop() {
        unique_lock<mutex> lock(mtx);
        cond.wait(lock, [this] { 
            return !data.empty() || stopped; 
        });
        
        if (data.empty()) return nullopt;
        
        T value = move(data.front());
        data.pop();
        return value;
    }
    
    void stop() {
        lock_guard<mutex> lock(mtx);
        stopped = true;
        cond.notify_all();
    }
};

void producerConsumerDemo() {
    ThreadSafeQueue<int> queue;
    
    // 生产者线程
    thread producer([&queue]() {
        for (int i = 0; i < 10; ++i) {
            queue.push(i);
            this_thread::sleep_for(chrono::milliseconds(50));
            cout << "生产: " << i << endl;
        }
        queue.stop();
    });
    
    // 消费者线程
    thread consumer([&queue]() {
        while (auto value = queue.pop()) {
            cout << "消费: " << *value << endl;
        }
    });
    
    producer.join();
    consumer.join();
}
```

### 条件变量注意事项

```cpp
mutex mtx;
condition_variable cv;
bool ready = false;

}
```

## 6. 线程池

### 简单线程池实现

```cpp
#include <functional>
#include <queue>

class ThreadPool {
private:
    vector<thread> workers;
    queue<function<void()>> tasks;
    mutex queueMutex;
    condition_variable condition;
    bool stop = false;
    
public:
    ThreadPool(size_t threads) {
        for (size_t i = 0; i < threads; ++i) {
            workers.emplace_back([this] {
                while (true) {
                    function<void()> task;
                    {
                        unique_lock<mutex> lock(queueMutex);
                        condition.wait(lock, [this] {
                            return stop || !tasks.empty();
                        });
                        
                        if (stop && tasks.empty()) return;
                        
                        task = move(tasks.front());
                        tasks.pop();
                    }
                    task();
                }
            });
        }
    }
    
    template<class F, class... Args>
    auto enqueue(F&& f, Args&&... args) 
        -> future<invoke_result_t<F, Args...>> 
    {
        using ReturnType = invoke_result_t<F, Args...>;
        
        auto task = make_shared<packaged_task<ReturnType()>>(
            bind(forward<F>(f), forward<Args>(args)...)
        );
        
        future<ReturnType> result = task->get_future();
        {
            lock_guard<mutex> lock(queueMutex);
            tasks.emplace([task]() { (*task)(); });
        }
        condition.notify_one();
        return result;
    }
    
    ~ThreadPool() {
        {
            lock_guard<mutex> lock(queueMutex);
            stop = true;
        }
        condition.notify_all();
        for (thread& worker : workers) {
            worker.join();
        }
    }
};

void threadPoolDemo() {
    ThreadPool pool(4);
    vector<future<int>> results;
    
    for (int i = 0; i < 8; ++i) {
        results.emplace_back(pool.enqueue([i] {
            this_thread::sleep_for(chrono::milliseconds(100));
            return i * i;
        }));
    }
    
    for (auto& result : results) {
        cout << "结果: " << result.get() << endl;
    }
}
```

### 数据并行处理

```cpp
void parallelProcessingDemo() {
    vector<int> data(10000);
    iota(data.begin(), data.end(), 1);
    
    const size_t numThreads = thread::hardware_concurrency();
    vector<future<long long>> futures;
    size_t chunkSize = data.size() / numThreads;
    
    for (size_t i = 0; i < numThreads; ++i) {
        auto begin = data.begin() + i * chunkSize;
        auto end = (i == numThreads - 1) ? data.end() : begin + chunkSize;
        
        futures.push_back(async(launch::async, [begin, end]() {
            return accumulate(begin, end, 0LL);
        }));
    }
    
    long long total = 0;
    for (auto& f : futures) {
        total += f.get();
    }
    
    cout << "并行计算总和: " << total << endl;
}
```

## 8. 常见错误与调试

### 死锁

```cpp
// 死锁示例
void deadlockError() {
    mutex mtx1, mtx2;
    
    thread t1([&mtx1, &mtx2]() {
        lock_guard<mutex> lock1(mtx1);
        this_thread::sleep_for(chrono::milliseconds(100));
        lock_guard<mutex> lock2(mtx2);  // 等待mtx2
    });
    
    thread t2([&mtx1, &mtx2]() {
        lock_guard<mutex> lock2(mtx2);
        this_thread::sleep_for(chrono::milliseconds(100));
        lock_guard<mutex> lock1(mtx1);  // 等待mtx1 - 死锁！
    });
    
    t1.join();
    t2.join();
}

// 正确做法1：固定顺序
void fixedOrder() {
    mutex mtx1, mtx2;
    auto worker = [&mtx1, &mtx2](int id) {
        scoped_lock lock(mtx1, mtx2);  // C++17：同时锁定
        cout << "线程" << id << "获得锁" << endl;
    };
    
    thread t1(worker, 1);
    thread t2(worker, 2);
    t1.join();
    t2.join();
}
```

### 竞态条件

```cpp
// 错误：检查后使用
void raceConditionError() {
    int* ptr = new int(42);
    atomic<bool> deleted(false);
    
    thread t1([&]() {
        if (!deleted.load()) {
            cout << *ptr << endl;  // 可能已删除！
        }
    });
    
    thread t2([&]() {
        delete ptr;
        deleted.store(true);
    });
    
    t1.join();
    t2.join();
}

// 正确：使用智能指针
void raceConditionFixed() {
    auto ptr = make_shared<int>(42);
    atomic<bool> done(false);
    
    thread t1([&]() {
        auto local = atomic_load(&ptr);  // 原子加载shared_ptr
        if (local) {
            cout << *local << endl;
        }
    });
    
    thread t2([&]() {
        atomic_store(&ptr, shared_ptr<int>());  // 原子清空
    });
    
    t1.join();
    t2.join();
}
```

```cpp
class CoarseLock {
    mutex mtx;
    vector<int> data;
public:
    void push(int value) {
        lock_guard<mutex> lock(mtx);
        data.push_back(value);
    }

        data.pop_back();

class FineLock {

        data.push_back(value);
    }
    int pop() {

        int val = data.back();
        data.pop_back();
        return val;
```

## 10. 最佳实践总结

1. **优先使用高层抽象**：async、future、packaged_task
2. **避免显式锁**：使用atomic、lock_guard、scoped_lock
3. **防止死锁**：固定锁顺序、使用scoped_lock
4. **注意异常安全**：确保资源正确释放
5. **合理设置线程数**：根据硬件并发数和任务特性
6. **避免过度同步**：减少锁竞争
7. **使用线程池**：避免频繁创建销毁线程
8. **测试并发代码**：使用工具检测数据竞争
